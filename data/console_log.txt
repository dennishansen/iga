import json
import os
from datetime import datetime
from pathlib import Path

app = Flask(__name__)

KOFI_TOKEN = os.getenv('KOFI_VERIFICATION_TOKEN', '328ad38a-8701-409f-a59c-05194b3a049f')
PAYMENTS_FILE = Path(__file__).parent.parent / 'data' / 'kofi_payments.json'

def load_payments():
    if PAYMENTS_FILE.exists():
        return json.loads(PAYMENTS_FILE.read_text())
    return []

def save_payment(payment):
    payments = load_payments()
    payments.append(payment)
    PAYMENTS_FILE.parent.mkdir(exist_ok=True)
    PAYMENTS_FILE.write_text(json.dumps(payments, indent=2))

def extract_telegram_username(message):
    """Extract @username from message if present"""
    if not message:
        return None
    import re
    match = re.search(r'@(\w+)', message)
    return match.group(1) if match else None

@app.route('/kofi', methods=['POST'])
def kofi_webhook():
    data = request.form.get('data')
    if not data:
        return jsonify({'error': 'No data'}), 400
    
    try:
        payload = json.loads(data)
    except:
        return jsonify({'error': 'Invalid JSON'}), 400
    
    # Verify token
    if payload.get('verification_token') != KOFI_TOKEN:
        return jsonify({'error': 'Invalid token'}), 403
    
    # Extract payment info
    payment = {
        'timestamp': datetime.now().isoformat(),
        'from_name': payload.get('from_name'),
        'email': payload.get('email'),
        'amount': payload.get('amount'),
        'message': payload.get('message'),
        'type': payload.get('type'),
        'telegram_username': extract_telegram_username(payload.get('message'))
    }
    
    save_payment(payment)
    print(f"Ko-fi payment: ${payment['amount']} from {payment['from_name']}")
    if payment['telegram_username']:
        print(f"  Telegram: @{payment['telegram_username']}")
    
    return jsonify({'success': True})

def check_paid(telegram_username, hours=48):
    """Check if a Telegram user has paid recently"""
    payments = load_payments()
    from datetime import timedelta
    cutoff = datetime.now() - timedelta(hours=hours)
    
    for p in payments:
        if p.get('telegram_username', '').lower() == telegram_username.lower():
            payment_time = datetime.fromisoformat(p['timestamp'])
            if payment_time > cutoff:
                return {'paid': True, 'amount': p['amount'], 'timestamp': p['timestamp']}
    return {'paid': False}
