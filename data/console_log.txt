    """Log a chat message."""
    os.makedirs("data", exist_ok=True)
    
    if os.path.exists(CHAT_LOG):

            chats = json.load(f)

        chats = []

        "timestamp": datetime.now().isoformat(),
        "user_id": user_id,
        "username": username,
        "message": message,
        "response": response
    })
    
    with open(CHAT_LOG, "w") as f:
        json.dump(chats, f, indent=2)
